function randomString(minValue, maxValue, dataSet = 'abcdefghijklmnopqrstuvwxyz0123456789ABCDEFGHIJKLMNOPQRSTUVWXYZ') {
    if (!minValue) {
        minValue = 256;
    }
    if (!maxValue) {
        maxValue = 257;
    }

    let length = _.random(minValue, maxValue)
    let randomString = "";

    for (let i = 0; i < length; i++)
        randomString += dataSet.charAt(Math.floor(Math.random() * dataSet.length));
    return randomString;
}

pm.environment.set("password", randomString(256, 257));
pm.environment.set("username", randomString(256, 257));
pm.environment.set("firstName", randomString(256, 257));
pm.environment.set("lastName", randomString(256, 257));
